🗄️ Database Basics

Databases store organized data. Tables contain rows and columns.

-- Create a database
CREATE DATABASE company;

-- Use the database
USE company;

-- Create a table
CREATE TABLE employees (
    id INT PRIMARY KEY,
    name VARCHAR(100),
    salary DECIMAL(10,2)
);

📝 CREATE & INSERT

Create tables and add data using INSERT statement.

CREATE TABLE products (
    product_id INT PRIMARY KEY,
    product_name VARCHAR(255),
    price DECIMAL(10,2)
);

INSERT INTO products (product_id, product_name, price)
VALUES (1, 'Laptop', 999.99),
       (2, 'Mouse', 29.99),
       (3, 'Keyboard', 79.99);

🔍 SELECT Statement

Retrieve data from database tables.

-- Select all columns
SELECT * FROM employees;

-- Select specific columns
SELECT name, salary FROM employees;

-- Select with alias
SELECT name AS employee_name, salary FROM employees;

🎯 WHERE Clause

Filter records based on conditions.

-- Equal condition
SELECT * FROM employees WHERE department = 'IT';

-- Comparison operators
SELECT * FROM products WHERE price > 50;

-- Multiple conditions
SELECT * FROM employees 
WHERE age >= 25 AND salary > 50000;

-- Pattern matching
SELECT * FROM customers WHERE name LIKE 'J%';

📊 ORDER BY & GROUP BY

Sort results and group rows for aggregation.

-- Sort results
SELECT name, salary FROM employees
ORDER BY salary DESC;

-- Group by with aggregation
SELECT department, COUNT(*) as emp_count, AVG(salary) as avg_salary
FROM employees
GROUP BY department;

-- Having clause (filter groups)
SELECT department, AVG(salary) as avg_salary
FROM employees
GROUP BY department
HAVING AVG(salary) > 60000;

🔗 JOIN Operations

Combine data from multiple tables.

-- INNER JOIN
SELECT employees.name, departments.dept_name
FROM employees
INNER JOIN departments ON employees.dept_id = departments.id;

-- LEFT JOIN
SELECT customers.name, orders.order_date
FROM customers
LEFT JOIN orders ON customers.id = orders.customer_id;

-- Multiple joins
SELECT e.name, d.dept_name, p.project_name
FROM employees e
JOIN departments d ON e.dept_id = d.id
JOIN projects p ON e.id = p.emp_id;

✏️ UPDATE & DELETE

Modify or remove existing data.

-- Update records
UPDATE employees
SET salary = salary * 1.10
WHERE department = 'Sales';

-- Delete records
DELETE FROM employees
WHERE status = 'inactive';

-- Be careful with DELETE (use WHERE clause!)
-- DELETE FROM employees; -- Removes ALL rows!

📈 Advanced Queries

Subqueries, views, and indexes for optimization.

-- Subquery
SELECT name, salary
FROM employees
WHERE salary > (SELECT AVG(salary) FROM employees);

-- Create view
CREATE VIEW high_earners AS
SELECT name, salary, department
FROM employees
WHERE salary > 75000;

-- Index for faster queries
CREATE INDEX idx_lastname ON customers(last_name);

-- Union of results
SELECT name FROM employees
UNION
SELECT name FROM contractors;
