🏗️ Classes & Objects (OOP)

Classes encapsulate data and functions. Objects are instances of classes.

class Car {
public:
    string brand;
    int year;
    
    void drive() {
        cout << brand << " is driving" << endl;
    }
};

int main() {
    Car myCar;
    myCar.brand = "Tesla";
    myCar.drive();
    return 0;
}

🌳 Inheritance

Derived classes inherit properties and methods from base classes.

class Vehicle {
public:
    void honk() {
        cout << "Beep beep!" << endl;
    }
};

class Car : public Vehicle {
public:
    string model;
};

int main() {
    Car myCar;
    myCar.honk();  // Inherited method
    return 0;
}

🎨 Polymorphism

Virtual functions allow derived classes to override base class methods.

class Animal {
public:
    virtual void sound() {
        cout << "Animal makes sound" << endl;
    }
};

class Dog : public Animal {
public:
    void sound() override {
        cout << "Dog barks" << endl;
    }
};

📊 Vectors & STL

Standard Template Library provides dynamic arrays, algorithms, and containers.

#include <vector>
#include <algorithm>

vector<int> numbers = {5, 2, 8, 1};
sort(numbers.begin(), numbers.end());

for(int num : numbers) {
    cout << num << " ";
}

⚠️ Exception Handling

Handle runtime errors gracefully using try-catch blocks.

try {
    int age;
    cout << "Enter age: ";
    cin >> age;
    
    if(age < 0) {
        throw invalid_argument("Age cannot be negative");
    }
} catch(const exception& e) {
    cout << "Error: " << e.what() << endl;
}

📄 File I/O

Read from and write to files using fstream library.

#include <fstream>

// Writing to file
ofstream outFile("data.txt");
outFile << "Hello, File!" << endl;
outFile.close();

// Reading from file
ifstream inFile("data.txt");
string line;
while(getline(inFile, line)) {
    cout << line << endl;
}
